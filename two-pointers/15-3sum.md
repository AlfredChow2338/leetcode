### Takeaways

1. Brute force and stringify array as unique identifier, but time limit exceeded

```jsx
var threeSum = function(nums) {
    const len = nums.length
    if (len < 3) {
        return []
    }
    const res = new Set()
    nums.sort((a, b) => a - b)
    for (let i = 0; i < len - 2; i++) {
        for (let j = i + 1; j < len - 1; j++) {
            for (let k = j + 1; k < len; k++) {
                if (nums[i] + nums[j] + nums[k] === 0) {
                    res.add(JSON.stringify([nums[i], nums[j], nums[k]]))
                }
            }
        }
    }
    return Array.from(res.values()).map(item => JSON.parse(item))
};
```

2. A common mistake is not properly skipping duplicate values, which leads to duplicate triplets in the result.

3. Without sorting, moving pointers based on sum comparisons does not guarantee you will find all valid triplets.

4. Two pointers
```jsx
var threeSum = function(nums) {
    const res = []
    nums.sort((a, b) => a - b)
    for (let i = 0; i < nums.length - 2; i++) {
        if (nums[i] > 0) {
            return res
        }
        if (i > 0 && nums[i-1] === nums[i]) {
            continue
        }
        let l = i + 1
            r = nums.length - 1
        while (l < r) {
            const sum = nums[i] + nums[l] + nums[r]
            if (sum > 0) {
                r --
                continue
            }
            if (sum < 0) {
                l ++
                continue
            }
            if (sum === 0) {
                res.push([nums[i], nums[l], nums[r]])
                l ++
                r --
                while (l < r && nums[l-1] === nums[l]) {
                    l ++
                }
            }
        }
    }
    return res
};
```

5. Hash Map: Use a hash map that stores how many times each number appears. As we pick the first and second numbers, we temporarily reduce their counts in the map so we don't reuse them. Then we check whether the needed third value still exists in the map.

```jsx
var threeSum = function(nums) {
    const cnt = new Map()
    const res = []
    nums.sort((a, b) => a - b)
    for (const n of nums) {
        cnt.set(n, cnt.has(n) ? cnt.get(n) + 1 : 1)
    } 
    for (let i = 0; i < nums.length - 2; i++) {
        if (nums[i] > 0) {
            return res
        }
        cnt.set(nums[i], cnt.get(nums[i]) - 1)
        if (i > 0 && nums[i-1] === nums[i]) {
            continue
        }
        for (j = i + 1; j < nums.length; j++) {
            cnt.set(nums[j], cnt.get(nums[j]) - 1)
            if (j > i + 1 && nums[j-1] === nums[j]) {
                continue
            }
            const target = 0 - nums[i] - nums[j]
            if (cnt.get(target) > 0) {
                res.push([nums[i], nums[j], target])
            }
        }
        for (j = i + 1; j < nums.length; j++) {
            cnt.set(nums[j], cnt.get(nums[j]) + 1)
        }
    }
    return res
};
```
